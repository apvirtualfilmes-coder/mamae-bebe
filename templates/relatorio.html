def carregar_dados(self): try: with open('dados_bebe.json', 'r') as f: return json.load(f) except: return {'nome_mae': '', 'nome_bebe': '', 'data_nasc': '', 'hora_nasc': '', 'peso_bebe': '', 'registros': []} def salvar_dados(self, instance): self.dados = { 'nome_mae': self.nome_mae_input.text, 'nome_bebe': self.nome_bebe_input.text, 'data_nasc': self.data_input.text, 'hora_nasc': self.hora_input.text, 'peso_bebe': self.peso_input.text, 'registros': self.registros } with open('dados_bebe.json', 'w') as f: json.dump(self.dados, f) self.nome_mae = self.nome_mae_input.text self.nome_bebe = self.nome_bebe_input.text self.data_nasc = self.data_input.text self.hora_nasc = self.hora_input.text self.peso_bebe = self.peso_input.text self.atualizar_tudo(None) self.mostrar_popup('Dados salvos com sucesso!') def calcular_idade(self): try: if not self.data_nasc or not self.hora_nasc: return 'Preencha os dados' nasc = datetime.strptime(self.data_nasc + ' ' + self.hora_nasc, '%d/%m/%Y %H:%M') diff = datetime.now() - nasc dias = diff.days if dias > 30: return f'{dias//30}m e {dias%30}d' elif dias > 0: return f'{dias}d' else: horas = diff.seconds // 3600 return f'{horas}h' except: return 'Preencha data/hora' # ===== MAMADA INTEGRADA ===== def iniciar_mamada(self, instance): if not self.nome_bebe: self.mostrar_popup('Cadastre o bebe primeiro!') return if not self.mamada_ativa: # INICIAR MAMADA self.mamada_ativa = True self.mamada_inicio = datetime.now() self.btn_mamar.text = 'PARAR MAMADA' self.btn_mamar.background_color = COR_VERMELHO self.mostrar_popup('Mamada iniciada! Clique em PARAR quando terminar.') else: # PARAR MAMADA - REGISTRA AUTOMATICAMENTE fim = datetime.now() duracao = fim - self.mamada_inicio minutos = int(duracao.total_seconds() // 60) segundos = int(duracao.total_seconds() % 60) tempo_str = f'{minutos:02d}:{segundos:02d}' agora = datetime.now().strftime('%d/%m/%Y %H:%M') self.registros.append({'tipo': 'Mamou', 'data': agora, 'duracao': tempo_str}) self.dados['registros'] = self.registros with open('dados_bebe.json', 'w') as f: json.dump(self.dados, f) self.mamada_ativa = False self.mamada_inicio = None self.btn_mamar.text = 'MAMAR' self.btn_mamar.background_color = COR_VERDE self.tempo_label.text = 'Tempo: 00:00' mae = self.nome_mae or 'Mamae' bebe = self.nome_bebe or 'bebe' msg = f'{mae}, {bebe} mamou por {tempo_str}!\n\n' if minutos < 5: msg += 'Dica: Tente oferecer mais tempo de mamada.' elif minutos >= 15: msg += 'Otimo tempo de mamada!' else: msg += 'Mamada com duracao adequada!' self.mostrar_popup(msg) self.atualizar_tudo(None) def atualizar_cronometro(self): if self.mamada_ativa and self.mamada_inicio: diff = datetime.now() - self.mamada_inicio minutos = int(diff.total_seconds() // 60) segundos = int(diff.total_seconds() % 60) self.tempo_label.text = f'Tempo: {minutos:02d}:{segundos:02d}' # ===== REGISTROS ===== def registrar_evento(self, tipo, msg): if not self.nome_bebe: self.mostrar_popup('Cadastre o bebe primeiro!') return agora = datetime.now().strftime('%d/%m/%Y %H:%M') self.registros.append({'tipo': tipo, 'data': agora}) self.dados['registros'] = self.registros with open('dados_bebe.json', 'w') as f: json.dump(self.dados, f) self.mostrar_popup(msg) self.atualizar_tudo(None) def registrar_coco(self, instance): mae = self.nome_mae or 'Mamae' bebe = self.nome_bebe self.registrar_evento('Fez Coco', f'{mae}, {bebe} fez coco! Cor amarela = saudavel.') def registrar_xixi(self, instance): mae = self.nome_mae or 'Mamae' bebe = self.nome_bebe self.registrar_evento('Fez Xixi', f'{mae}, {bebe} fez xixi! Continue hidratando.') def registrar_regurgito(self, instance): mae = self.nome_mae or 'Mamae' bebe = self.nome_bebe self.registrar_evento('Regurgitou', f'{mae}, {bebe} regurgitou. Mantenha em pe apos mamar.') def registrar_sono(self, instance): mae = self.nome_mae or 'Mamae' bebe = self.nome_bebe self.registrar_evento('Dormiu', f'{mae}, {bebe} dormiu! O sono e essencial.') # ===== VERIFICAR LEMBRETE ===== def verificar_lembrete(self, dt): if not self.nome_bebe or not self.registros: return hoje = datetime.now().strftime('%d/%m/%Y') registros_hoje = [r for r in self.registros if r['data'].startswith(hoje)] mamadas = [r for r in registros_hoje if r['tipo'] == 'Mamou'] if mamadas: ultima = datetime.strptime(mamadas[-1]['data'], '%d/%m/%Y %H:%M') diff = datetime.now() - ultima if diff.total_seconds() > 10800: # 3 horas mae = self.nome_mae or 'Mamae' bebe = self.nome_bebe self.mostrar_popup(f'{mae}, ja faz 3h desde a ultima mamada de {bebe}. Hora de mamar!') # ===== VACINAS ===== def mostrar_vacinas(self, instance): texto = 'CALENDARIO DE VACINAS\n\n' texto += 'Consulte o pediatra para confirmar!\n\n' vacinas = [ 'BCG - Ao nascer', 'Hepatite B - Ao nascer', 'Penta - 2, 4, 6 meses', 'VIP - 2, 4, 6 meses', 'Pneumococica - 2, 4, 6 meses', 'Rotavirus - 2, 4 meses', 'Meningococica C - 3, 5 meses', 'Triviral Viral - 12 meses' ] for v in vacinas: texto += f'- {v}\n' self.mostrar_popup(texto) # ===== RELATORIO ===== def mostrar_relatorio(self, instance): if not self.registros: self.mostrar_popup('Nenhum registro ainda!') return hoje = datetime.now().strftime('%d/%m/%Y') registros_hoje = [r for r in self.registros if r['data'].startswith(hoje)] mae = self.nome_mae or 'Mamae' bebe = self.nome_bebe or 'Bebe' texto = f'RELATORIO - {hoje}\n\n' texto += f'Mae: {mae}\n' texto += f'Bebe: {bebe}\n' texto += f'Peso: {self.peso_bebe or "--"}kg\n' texto += '=' * 30 + '\n\n' if registros_hoje: mamadas = len([r for r in registros_hoje if r['tipo'] == 'Mamou']) coco = len([r for r in registros_hoje if r['tipo'] == 'Fez Coco']) xixi = len([r for r in registros_hoje if r['tipo'] == 'Fez Xixi']) sono = len([r for r in registros_hoje if r['tipo'] == 'Dormiu']) regurgito = len([r for r in registros_hoje if r['tipo'] == 'Regurgitou']) texto += f'Mamadas: {mamadas}x\n' texto += f'Coco: {coco}x\n' texto += f'Xixi: {xixi}x\n' texto += f'Sono: {sono}x\n' texto += f'Regurgito: {regurgito}x\n\n' texto += 'AVALIACAO:\n' if mamadas >= 8: texto += 'Amamentacao: OTIMA\n' elif mamadas >= 5: texto += 'Amamentacao: REGULAR\n' else: texto += 'Amamentacao: BAIXA\n' if xixi >= 6: texto += 'Hidratacao: OTIMA\n' else: texto += 'Hidratacao: BAIXA\n' else: texto += 'Nenhuma atividade registrada hoje.' self.mostrar_popup(texto) def atualizar_tudo(self, dt): self.atualizar_cronometro() self.atualizar_contadores() self.atualizar_status() self.atualizar_ultimos_registros() nome = self.nome_bebe or 'Bebe' idade = self.calcular_idade() peso = self.peso_bebe or '--' self.idade_label.text = f'{nome} - {idade} | Peso: {peso}kg' def atualizar_contadores(self): hoje = datetime.now().strftime('%d/%m/%Y') registros_hoje = [r for r in self.registros if r['data'].startswith(hoje)] mamadas = len([r for r in registros_hoje if r['tipo'] == 'Mamou']) self.total_mamadas_label.text = str(mamadas) # Ultima mamada mamadas_hoje = [r for r in registros_hoje if r['tipo'] == 'Mamou'] if mamadas_hoje: ultima = mamadas_hoje[-1]['data'].split()[1] self.ultima_mamada_label.text = f'Ultima mamada: {ultima}' else: self.ultima_mamada_label.text = 'Ultima mamada: --' def atualizar_status(self): if not self.nome_bebe or not self.registros: self.status_label.text = 'Cadastre o bebe e comece a monitorar!' return hoje = datetime.now().strftime('%d/%m/%Y') registros_hoje = [r for r in self.registros if r['data'].startswith(hoje)] mamadas = len([r for r in registros_hoje if r['tipo'] == 'Mamou']) xixi = len([r for r in registros_hoje if r['tipo'] == 'Fez Xixi']) status = [] if mamadas >= 8: status.append('Amamentacao: OTIMA') elif mamadas >= 5: status.append('Amamentacao: REGULAR') else: status.append('Amamentacao: BAIXA') if xixi >= 6: status.append('Hidratacao: OTIMA') elif xixi >= 3: status.append('Hidratacao: REGULAR') else: status.append('Hidratacao: BAIXA') self.status_label.text = ' | '.join(status) def atualizar_ultimos_registros(self): if self.registros: ultimos = self.registros[-5:][::-1] texto = '\n'.join([f'- {r["tipo"]} - {r["data"]}' for r in ultimos]) self.ultimos_label.text = texto else: self.ultimos_label.text = 'Nenhum registro ainda' def mostrar_popup(self, texto): layout = BoxLayout(orientation='vertical', spacing=10, padding=20) label = Label(text=texto, text_size=(380, None), halign='left', color=COR_TEXTO, font_size='14sp') layout.add_widget(label) btn = Button(text='Fechar', background_color=COR_ROSA, color=COR_BRANCO, bold=True, size_hint_y=None, height=45) layout.add_widget(btn) popup = Popup(title='', content=layout, size_hint=(0.9, 0.7)) btn.bind(on_press=popup.dismiss) popup.open() if __name__ == '__main__': MamaApp().run()
